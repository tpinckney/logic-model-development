    <script>

            const addItemBtn = document.getElementById('add-item-btn');

            const clearAllBtn = document.getElementById('clear-all-btn');

            const exportBtn = document.getElementById('export-btn');

            const closeModalBtn = document.getElementById('close-modal-btn');

            const copyBtn = document.getElementById('copy-btn');

            const exportModal = document.getElementById('export-modal');

            const exportContent = document.getElementById('export-content');

            

            // Add new item

            addItemBtn.addEventListener('click', function() {

                const itemText = document.getElementById('item-text').value.trim();

                const itemCategory = document.getElementById('item-category').value;

                

                if (itemText) {

                    const listId = itemCategory === 'short-outcomes' ? 'short-outcomes-list' : 

                                  itemCategory === 'medium-outcomes' ? 'medium-outcomes-list' :

                                  itemCategory === 'long-outcomes' ? 'long-outcomes-list' : 

                                  `${itemCategory}-list`;

                    

                    const list = document.getElementById(listId);

                    const newItem = document.createElement('li');

                    newItem.className = 'bg-brand-cream p-2 rounded shadow-sm flex justify-between items-center';

                    newItem.innerHTML = `

                        <span>${itemText}</span>

                        <button class="delete-btn text-red-700 hover:text-red-900" aria-label="Delete item">

                            <svg xmlns="http://www.w3.org/2000/svg" class="h-4 w-4" fill="none" viewBox="0 0 24 24" stroke="currentColor">

                                <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M6 18L18 6M6 6l12 12" />

                            </svg>

                        </button>

                    `;

                    list.appendChild(newItem);

                    

                    // Add delete functionality to the new item

                    const deleteBtn = newItem.querySelector('.delete-btn');

                    deleteBtn.addEventListener('click', function() {

                        newItem.remove();

                    });

                    

                    // Clear input field

                    document.getElementById('item-text').value = '';

                }

            });

            

            // Add delete functionality to existing items

            document.querySelectorAll('#inputs-list li, #activities-list li, #outputs-list li, #short-outcomes-list li, #medium-outcomes-list li, #long-outcomes-list li').forEach(item => {

                const text = item.textContent;

                item.innerHTML = `

                    <span>${text}</span>

                    <button class="delete-btn text-red-700 hover:text-red-900" aria-label="Delete item">

                        <svg xmlns="http://www.w3.org/2000/svg" class="h-4 w-4" fill="none" viewBox="0 0 24 24" stroke="currentColor">

                            <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M6 18L18 6M6 6l12 12" />

                        </svg>

                    </button>

                `;

                

                const deleteBtn = item.querySelector('.delete-btn');

                deleteBtn.addEventListener('click', function() {

                    item.remove();

                });

            });

            

            // Clear all items

            clearAllBtn.addEventListener('click', function() {

                if (confirm('Are you sure you want to clear all items?')) {

                    document.querySelectorAll('#inputs-list, #activities-list, #outputs-list, #short-outcomes-list, #medium-outcomes-list, #long-outcomes-list').forEach(list => {

                        list.innerHTML = '';

                    });

                }

            });

            

            // Export logic model

            exportBtn.addEventListener('click', function() {

                const inputs = Array.from(document.querySelectorAll('#inputs-list li')).map(li => li.querySelector('span').textContent);

                const activities = Array.from(document.querySelectorAll('#activities-list li')).map(li => li.querySelector('span').textContent);

                const outputs = Array.from(document.querySelectorAll('#outputs-list li')).map(li => li.querySelector('span').textContent);

                const shortOutcomes = Array.from(document.querySelectorAll('#short-outcomes-list li')).map(li => li.querySelector('span').textContent);

                const mediumOutcomes = Array.from(document.querySelectorAll('#medium-outcomes-list li')).map(li => li.querySelector('span').textContent);

                const longOutcomes = Array.from(document.querySelectorAll('#long-outcomes-list li')).map(li => li.querySelector('span').textContent);

                

                const exportHTML = `

                    <div class="mb-6 p-4 bg-brand-white rounded-lg border-l-4 border-brand-darkOlive">

                        <h3 class="text-lg font-semibold text-brand-darkOlive mb-2">Inputs (resources invested):</h3>

                        <ul class="list-disc pl-5">

                            ${inputs.map(input => `<li>${input}</li>`).join('')}

                        </ul>

                    </div>

                    <div class="mb-6 p-4 bg-brand-white rounded-lg border-l-4 border-brand-darkGold">

                        <h3 class="text-lg font-semibold text-brand-darkGold mb-2">Activities (what you do):</h3>

                        <ul class="list-disc pl-5">

                            ${activities.map(activity => `<li>${activity}</li>`).join('')}

                        </ul>

                    </div>

                    <div class="mb-6 p-4 bg-brand-white rounded-lg border-l-4 border-brand-darkOrange">

                        <h3 class="text-lg font-semibold text-brand-darkOrange mb-2">Outputs (immediate deliverables):</h3>

                        <ul class="list-disc pl-5">

                            ${outputs.map(output => `<li>${output}</li>`).join('')}

                        </ul>

                    </div>

                    <div class="mb-6 p-4 bg-brand-white rounded-lg border-l-4 border-brand-darkGold">

                        <h3 class="text-lg font-semibold text-brand-darkGold mb-2">Short-term Outcomes (knowledge, skills, awareness):</h3>

                        <ul class="list-disc pl-5">

                            ${shortOutcomes.map(outcome => `<li>${outcome}</li>`).join('')}

                        </ul>

                    </div>

                    <div class="mb-6 p-4 bg-brand-white rounded-lg border-l-4 border-brand-darkOrange">

                        <h3 class="text-lg font-semibold text-brand-darkOrange mb-2">Medium-term Outcomes (behavior, practice, performance):</h3>

                        <ul class="list-disc pl-5">

                            ${mediumOutcomes.map(outcome => `<li>${outcome}</li>`).join('')}

                        </ul>

                    </div>

                    <div class="mb-6 p-4 bg-brand-white rounded-lg border-l-4 border-brand-darkOlive">

                        <h3 class="text-lg font-semibold text-brand-darkOlive mb-2">Long-term Outcomes (impact, systems, sustainability):</h3>

                        <ul class="list-disc pl-5">

                            ${longOutcomes.map(outcome => `<li>${outcome}</li>`).join('')}

                        </ul>

                    </div>

                    <div class="text-sm text-gray-700 text-right">

                        Generated by Vision Shift Catalyst Logic Model Builder

                    </div>

                `;

                

                exportContent.innerHTML = exportHTML;

                exportModal.classList.remove('hidden');

            });

            

            // Close modal

            closeModalBtn.addEventListener('click', function() {

                exportModal.classList.add('hidden');

            });

            

            // Copy to clipboard

            copyBtn.addEventListener('click', function() {

                const range = document.createRange();

                range.selectNode(exportContent);

                window.getSelection().removeAllRanges();

                window.getSelection().addRange(range);

                document.execCommand('copy');

                window.getSelection().removeAllRanges();

                

                const originalText = copyBtn.textContent;

                copyBtn.textContent = 'Copied!';

                setTimeout(() => {

                    copyBtn.textContent = originalText;

                }, 2000);

            });

            

            // Close modal when clicking outside

            window.addEventListener('click', function(event) {

                if (event.target === exportModal) {

                    exportModal.classList.add('hidden');

                }

            });

            

            // Allow pressing Enter to add item

            document.getElementById('item-text').addEventListener('keypress', function(e) {

                if (e.key === 'Enter') {

                    addItemBtn.click();

                }

            });

 

            // Accessibility improvements

            document.getElementById('item-text').setAttribute('aria-label', 'Enter new item text');

            document.getElementById('item-category').setAttribute('aria-label', 'Select category for new item');

            document.getElementById('add-item-btn').setAttribute('aria-label', 'Add new item to logic model');

            document.getElementById('clear-all-btn').setAttribute('aria-label', 'Clear all items from logic model');

            document.getElementById('export-btn').setAttribute('aria-label', 'Export logic model');

            document.getElementById('copy-btn').setAttribute('aria-label', 'Copy logic model to clipboard');

        });
